#include <iostream>
#include <fstream>
#include <string>

using namespace std;

unsigned long mod47(string name)
{
    unsigned long groupNumber = 1;
    for (int i = 0; i < 6 && i < name.length(); i++) {
        groupNumber = groupNumber * (name[i] - 64);
    }
    
    return groupNumber % 47;
}

int main() {
    ofstream fout ("ride.out");
    ifstream fin ("ride.in");
    string a, b;
    fin >> a >> b;
    
    if (mod47(a) == mod47(b)) {
        fout << "GO" << endl;
    } else {
        fout << "STAY" << endl;
    }
    
    return 0;
}
